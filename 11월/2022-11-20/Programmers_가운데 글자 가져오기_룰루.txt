import java.util.*;

class Solution {
    public String solution(String s) {
        String answer = "";
        
        String[] arr = s.split("");
        int len = arr.length;
        
        if(len % 2 == 0){ //Â¦¼ö
            answer += arr[(len/2)-1];
            answer += arr[len/2];
            
        }else{ //È¦¼ö
            answer += arr[len/2];
        }
        
        return answer;
    }
}
